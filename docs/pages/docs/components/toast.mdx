# Toast

A concise, temporary message.

## Usage

First of all, you need to import the `useToasts` hook from the `kitchn` package.

```js
import { useToasts } from "kitchn"
```

## Default

<Playground
  code={`
() => {
  const { setToast } = useToasts();
  return (
    <Button
    onClick={() => {
    setToast({
      text: "The Evil Rabbit jumped over the fence.",
      delay: 3000,
        });
      }}
    >
      Show Toast
    </Button>
  );
}
`}
/>

### Actions

<Playground
  code={`
  () => {
  const { setToast } = useToasts();

  const alertAction = {
    name: "alert",
    handler: () => alert("alert from toast"),
  };

  return (
    <Button
      onClick={() => {
        setToast({
        text: "The Evil Rabbit jumped over the fence.",
        actions: [alertAction],
        });
      }}
    >
      Show Toast
    </Button>
  );
}
`}
/>

## Cancel

<Playground
  code={`
  () => {
  const { setToast } = useToasts();

  const alertAction = {
    name: "alert",
    handler: () => alert("alert from toast"),
  };

  const cancelAction = {
    name: "cancel",
    passive: true,
    handler: (_, cancel) => cancel(),
  };

  return (
    <Button
      onClick={() => {
        setToast({
        text: "The Evil Rabbit",
        actions: [alertAction, cancelAction],
        });
      }}
    >
      Show Toast
    </Button>
  );
}
`}
/>
