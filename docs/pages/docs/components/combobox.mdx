Combobox
A composable combobox component that combines an input field with a dropdown list of suggestions, providing auto-completion functionality and keyboard navigation support.
Import
jsxCopyimport { Combobox } from "kitchn";
Usage
The Combobox component consists of multiple subcomponents that can be composed together to create a customized auto-complete experience:
jsxCopy<Combobox>
  <Combobox.Input placeholder="Search..." />
  <Combobox.List>
    <Combobox.Option value="option1">Option 1</Combobox.Option>
    <Combobox.Option value="option2">Option 2</Combobox.Option>
  </Combobox.List>
</Combobox>
Examples
Basic Usage
<Playground
code={`
() => {
const [value, setValue] = useState("");
const options = ["Apple", "Banana", "Orange", "Pear"];
return (
<Combobox value={value} onChange={setValue}>
<Combobox.Input placeholder="Select a fruit..." />
<Combobox.List>
{options.map((option) => (
<Combobox.Option key={option} value={option}>
{option}
</Combobox.Option>
))}
</Combobox.List>
</Combobox>
);
}
`}
/>
With Custom Filtering
<Playground
code={`
() => {
const [value, setValue] = useState("");
const options = ["React", "Vue", "Angular", "Svelte"];
const filterOptions = (inputValue) => {
return options.filter(option =>
option.toLowerCase().includes(inputValue.toLowerCase())
);
};
return (
<Combobox 
   value={value} 
   onChange={setValue}
   onFilter={filterOptions}
 >
<Combobox.Input placeholder="Search frameworks..." />
<Combobox.List>
{options.map((option) => (
<Combobox.Option key={option} value={option}>
{option}
</Combobox.Option>
))}
</Combobox.List>
</Combobox>
);
}
`}
/>
